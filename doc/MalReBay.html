  imported_data$late_failures, 
  Patient.ID = gsub(" (Day 0|Day Failure)$", "", Sample.ID)

  alleles_with_patient_id, 
  match_results_df, 
  by = c("Patient.ID" = "Sample.ID")
)

  final_table, 
  mcmc_summary_for_merge, 
  by = c("Patient.ID" = "Sample.ID")
)
